function formatDuration (seconds) {

if (seconds === 0) {
return 'now'
}

let y = Math.floor(seconds / 31536000)
let d = seconds % 86400 !== 0 ? Math.floor(seconds / 86400) % 365 : 0
let h = Math.floor(seconds / 3600) !== 3600 ? Math.floor(seconds / 3600) % 24 : 0
let m = Math.floor(seconds / 60) !== 60 ? Math.floor(seconds / 60) % 60: 0
let s = seconds % 60

let arr = new Array(y, d, h, m, s)

let dateTranslate = [];
let dateName = {
                years: 'years',
                year: 'year',
                days: 'days',
                day: 'day',
                hours: 'hours',
                hour: 'hour',
                minutes: 'minutes',
                minute: 'minute',
                seconds: 'seconds',
                second: 'second'
                }

arr[0] == 0 ? dateTranslate : arr[0] == 1 ? dateTranslate.push(`${arr[0]} ${dateName.year}`) :  dateTranslate.push(`${arr[0]} ${dateName.years}`);
arr[1] == 0 ? dateTranslate : arr[1] == 1 ? dateTranslate.push(`${arr[1]} ${dateName.day}`) :  dateTranslate.push(`${arr[1]} ${dateName.days}`);
arr[2] == 0 ? dateTranslate : arr[2] == 1 ? dateTranslate.push(`${arr[2]} ${dateName.hour}`) :  dateTranslate.push(`${arr[2]} ${dateName.hours}`);
arr[3] == 0 ? dateTranslate : arr[3] == 1 ? dateTranslate.push(`${arr[3]} ${dateName.minute}`) :  dateTranslate.push(`${arr[3]} ${dateName.minutes}`);
arr[4] == 0 ? dateTranslate : arr[4] == 1 ? dateTranslate.push(`${arr[4]} ${dateName.second}`) :  dateTranslate.push(`${arr[4]} ${dateName.seconds}`);

return dateTranslate.length == 1 ? dateTranslate.join('') : dateTranslate.join(', ').replace(/,(?=[^,]*$)/, ' and')
}
